# 1. Set Pineapple monitor to Channel 9

# 2. Run 60s airodump-ng capture on Channel 9

  "rm -f /tmp/ch9-01.* 2>/dev/null; \
   airodump-ng wlan1mon --channel 9 --write /tmp/ch9 --output-format csv >/dev/null 2>&1 & \
   sleep 60; killall airodump-ng 2>/dev/null; sleep 2; cat /tmp/ch9-01.csv"

# 3. Filter results for target BSSIDs
# AP section: grep for 92:07:1D:BC:0E:C4, 92:07:1D:BC:0E:C5
# Client section: check BSSID column for matches
